```verilog
module SystemX(
    input wire A,
    input wire B,
    input wire C,
    input wire D,
    output wire Z
    );
    
    always @(A or B or C or D)
    begin
        wire n1, n2, n3, E, F;
        #5 n1 = A & B & C;
        #5 E = n1 | D;
        #5 n2 = !(B | C);
        #5 F = !(A & n2);
        #2 n3 = !F;
        #5 Z = (E | n3)&(!E | !n3);
    end
endmodule
```

```verilog
module SystemX(
    input integer C,
    input wire B1,
    input wire B2,
    input wire B3,
    output wire A
    );
    
    always @(B1 or B2 or B3 or C)
    begin
    if(C == 1)
        begin
            A = B1;
        end
    else if (C == 2)
        begin
            A = B2;
        end
    else if (C == 3)
        begin
            A = B3;
        end
    else
        begin
            A = 0;
        end
    end
endmodule
```


```verilog
module SystemX(
    input integer C,
    input wire B1,
    input wire B2,
    input wire B3,
    output wire A
    );
    
    assign A = C == 1? B1: C == 2? B2: C == 3? B3: 0;

endmodule
```
